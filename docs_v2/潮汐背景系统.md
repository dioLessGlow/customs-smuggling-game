## 四、核心元素详解

### 4.1 天空渐变

```javascript
// 根据当前小时插值计算颜色
function getSkyColor(hour) {
  const frame = Math.floor(hour / 24 * 9);  // 0-8
  const nextFrame = (frame + 1) % 9;
  const t = (hour % (24/9)) / (24/9);       // 插值因子

  return {
    top: lerpColor(PALETTE[frame].top, PALETTE[nextFrame].top, t),
    bottom: lerpColor(PALETTE[frame].bottom, PALETTE[nextFrame].bottom, t)
  };
}

// 绘制渐变
const gradient = ctx.createLinearGradient(0, 0, 0, height);
gradient.addColorStop(0, skyColor.top);
gradient.addColorStop(1, skyColor.bottom);
ctx.fillStyle = gradient;
ctx.fillRect(0, 0, width, height);
```

### 4.2 星星系统

```javascript
// 140颗星星，随机位置+大小+闪烁相位
const stars = Array.from({ length: 140 }, () => ({
  x: Math.random() * width,
  y: Math.random() * height * 0.6,  // 只在上半部分
  size: Math.random() * 2 + 0.5,
  phase: Math.random() * Math.PI * 2,  // 闪烁相位
  speed: Math.random() * 0.02 + 0.01
}));

// 绘制
stars.forEach(star => {
  const alpha = (Math.sin(time * star.speed + star.phase) + 1) / 2 * starOpacity;
  ctx.fillStyle = `rgba(255, 255, 255, ${alpha})`;
  ctx.beginPath();
  ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2);
  ctx.fill();
});
```

### 4.3 太阳/月亮

```javascript
// 位置计算（水平移动+弧形升降）
const sunX = width * 0.1 + (hour / 24) * width * 0.8;
const sunY = height * 0.3 + Math.sin((hour / 24) * Math.PI) * height * 0.2;

// 白天显示太阳，夜间显示月亮
if (hour >= 6 && hour <= 18) {
  // 太阳：金黄色，发光效果
  drawSun(sunX, sunY, '#ffd700');
} else {
  // 月亮：银白色，月牙形状
  drawMoon(sunX, sunY, '#c0c0c0');
}
```

### 4.4 波浪系统（26层）

```javascript
// 26层波浪，每层不同参数
const waves = Array.from({ length: 26 }, (_, i) => ({
  y: height * 0.5 + i * 15,           // 基础Y位置
  amplitude: 5 + i * 2,               // 振幅
  frequency: 0.01 + i * 0.002,        // 频率
  speed: 0.02 + i * 0.005,            // 速度
  opacity: 0.1 + (26 - i) * 0.03,     // 透明度（越远越淡）
  color: waveColor                      // 随时间变化的色调
}));

// 绘制单层波浪
function drawWave(wave, time) {
  ctx.beginPath();
  ctx.moveTo(0, wave.y);

  for (let x = 0; x <= width; x += 5) {
    const y = wave.y + Math.sin(x * wave.frequency + time * wave.speed) * wave.amplitude;
    ctx.lineTo(x, y);
  }

  ctx.lineTo(width, height);
  ctx.lineTo(0, height);
  ctx.closePath();

  ctx.fillStyle = `rgba(${waveColor.r}, ${waveColor.g}, ${waveColor.b}, ${wave.opacity})`;
  ctx.fill();
}
```

### 4.5 闪光系统（220个）

```javascript
// 波光粼粼效果
const sparkles = Array.from({ length: 220 }, () => ({
  x: Math.random() * width,
  y: height * 0.5 + Math.random() * height * 0.5,
  size: Math.random() * 3,
  phase: Math.random() * Math.PI * 2,
  speed: Math.random() * 0.05 + 0.02
}));

// 绘制
sparkles.forEach(sparkle => {
  const alpha = Math.sin(time * sparkle.speed + sparkle.phase);
  if (alpha > 0.5) {
    ctx.fillStyle = `rgba(255, 255, 255, ${alpha})`;
    ctx.beginPath();
    ctx.arc(sparkle.x, sparkle.y, sparkle.size, 0, Math.PI * 2);
    ctx.fill();
  }
});
```

### 5.2 主循环

```javascript
function render(time) {
  const hour = (Date.now() / 1000 / 60 / 60) % 24;  // 实时小时

  // 1. 清空画布
  ctx.clearRect(0, 0, width, height);

  // 2. 绘制天空渐变
  drawSky(hour);

  // 3. 绘制星星（夜间）
  if (hour < 6 || hour > 18) {
    drawStars(time, hour);
  }

  // 4. 绘制太阳/月亮
  drawCelestialBody(hour);

  // 5. 绘制云层
  drawClouds(time);

  // 6. 绘制海鸥
  drawSeagulls(time);

  // 7. 绘制波浪（26层）
  drawWaves(time, hour);

  // 8. 绘制闪光
  drawSparkles(time);

  // 9. 绘制暗角
  drawVignette();

  // 10. 绘制时间UI
  drawTimeUI(hour);

  requestAnimationFrame(render);
}
```

## 六、DPR适配

```javascript
const dpr = window.devicePixelRatio || 1;

// Canvas尺寸 × dpr
canvas.width = 390 * dpr;
canvas.height = 844 * dpr;

// CSS尺寸保持逻辑分辨率
canvas.style.width = '390px';
canvas.style.height = '844px';

// 坐标系缩放
ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

// 之后所有绘制使用逻辑坐标（0-390, 0-844）
```

## 八、性能优化

| 优化点 | 策略 |
|--------|------|
| 波浪绘制 | 每5像素一个点，减少lineTo调用 |
| 星星/闪光 | 只绘制alpha>0.1的，跳过不可见的 |
| 云层 | 使用简单圆形组合，避免复杂路径 |
| 缓存 | 调色板颜色预计算，避免每帧插值 |
| 帧率 | 与屏幕刷新率同步（requestAnimationFrame） |
