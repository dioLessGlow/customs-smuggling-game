```
    ↓
解锁导师"私教课"界面
    ↓
导师对话开场 → 案例讲解 → 数据展示 → 法律依据
    ↓
课后小测验（3道选择题，每道3选1）
    ↓
    ├─ 3/3 全对 → +5功勋 + 解锁该关卡隐藏线索
    ├─ 2/3 对 → +2功勋
    └─ ≤1/3 → 无额外奖励，可重新学习
    ↓
永久效果激活（对应关卡属性加成）
```

## 八、购买逻辑代码

```javascript
function purchaseItem(itemId) {
  const item = SHOP_CONFIG[itemId];
  const save = SaveManager.load();

  // 1. 检查是否已拥有
  if (save.shopItems.includes(itemId) || save.shopKnowledge.includes(itemId)) {
    showToast('已拥有该商品');
    return;
  }

  // 2. 检查前置条件
  if (item.prerequisite && !save.shopItems.includes(item.prerequisite)) {
    showToast(`需要先购买: ${item.prerequisite_name}`);
    return;
  }

  // 3. 检查功勋是否足够
  if (save.permanentDao < item.cost) {
    const diff = item.cost - save.permanentDao;
    showToast(`功勋不足，还差 ${diff} 功勋！继续缉私任务获取吧！`);
    return;
  }

  // 4. 弹出确认框
  showConfirmDialog({
    title: '确认购买？',
    icon: item.icon,
    name: item.name,
    desc: item.description,
    cost: item.cost,
    remaining: save.permanentDao - item.cost,
    onConfirm: () => {
      // 5. 扣除功勋
      save.permanentDao -= item.cost;

      // 6. 标记已拥有
      if (item.category === 'knowledge') {
        save.shopKnowledge.push(itemId);
      } else {
        save.shopItems.push(itemId);
      }

      // 7. 应用效果
      applyItemEffect(item);

      // 8. 播放成功动画
      playSuccessAnimation(item);

      // 9. 保存存档
      SaveManager.save(save);

      // 10. 知识类打开私教课
      if (item.category === 'knowledge') {
        openKnowledgeLesson(itemId);
      }
    }
  });
}
```
